          SET {firstCharacter} = first character of the string
          SET {newString} equal "";

            IF {newString} is empty
              THEN {newString} equal {firstCharacter}
            ELSE 
              IF current character equals {firstCharacter}
                THEN newString += *
              ELSE 
                THEN newString += current character
